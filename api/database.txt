# app.quickdatabasediagrams.com
user
-
id int(11) PK IDENTITY
name varchar(50)
email varchar(50)
password varchar(50) NULL
key varchar(50)
role_id int(11) FK >- role.id
intro boolean default=FALSE

role
-
id int(11) PK IDENTITY 
level int(2)
name varchar(50)
page_access json

course
-
id int(11) PK IDENTITY 
title varchar(50)
discription text
data json

course_progression
-
id int(11) PK IDENTITY 
user_id int(11) FK >- user.id
course_id int(11) FK >- course.id
course_date_finish date NULL
progression json

event
-
id int(11) PK IDENTITY 
title varchar(50)
discription text
date datetime

event_voting
-
id int(11) PK IDENTITY 
event_id int(11) FK >- event.id
user_id int(11) FK >- user.id
